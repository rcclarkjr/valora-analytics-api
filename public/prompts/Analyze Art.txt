    // ✅ SMART HOME NAVIGATION - Routes to correct home page based on membership level
    function returnToMainMenu() {
      const membershipLevel = sessionStorage.getItem('membershipLevel');
      
      if (membershipLevel === 'pro') {
        window.location.href = 'https://advisory.valoraanalytics.com/products/the-artisan-s-ascent-pro-studio';
      } else {

    // UPDATED: Build clean HTML from JSON response with new format
    function buildAnalysisHTML(data, imageUrl, timestamp) {
      let html = `
        <div id="analysisText">
          <div class="report-title">Analysis: 33 Essential Factors</div>
          <div id="resultDate">As of ${timestamp}</div>
          <div class="artwork-info">
            <img src="${imageUrl}" alt="${data.artTitle}" class="artwork-image">
            <h1>Title: ${data.artTitle}</h1>
            <h3>Artist: ${data.artistName}</h3>
          </div>
          <div class="analysis-report">
      `;

      // Overview section
      if (data.overview) {
        html += `
          <div class="overview-section">
            <h3>Overview</h3>
            <p>${data.overview}</p>
          </div>
        `;
      }

      // Strengths section
      if (data.strengths && data.strengths.length > 0) {
        html += `<div class="strengths-section">
          <div class="section-title">Strengths</div>`;
        
        data.strengths.forEach((strength, index) => {
          html += `
            <div class="strength-item">
              <div class="strength-title">${index + 1}. ${strength.title}</div>
              <div class="strength-description">${strength.description}</div>
            </div>
          `;
        });
        
        html += `</div>`;
      }

      // Opportunities for Refinement section (UPDATED FORMAT)
      if (data.opportunities && data.opportunities.length > 0) {
        html += `<div class="opportunities-section">
          <div class="section-title">Opportunities for Refinement</div>
          <div class="opportunities-subheading">Only apply the options which resonate with your artistic voice.</div>`;
        
        data.opportunities.forEach((opportunity, categoryIndex) => {
          html += `
            <div class="opportunity-category">
              <div class="category-title">${opportunity.category}</div>
          `;
          
          if (opportunity.steps && opportunity.steps.length > 0) {
            html += `<div class="category-steps">`;
            
            opportunity.steps.forEach(step => {
              html += `
                <div class="step-item">
                  <span class="step-description">• ${step.description}</span>
                </div>
              `;
            });
            
            html += `</div>`;
          }
          
          html += `</div>`;
          
          // Add blank line before next category (except for last one)
          if (categoryIndex < data.opportunities.length - 1) {
            html += `<br>`;
          }
        });
        
        html += `</div>`;
      }

      // Factors Suggested For Further Study section (UPDATED HEADING)
      if (data.recommendedStudy && data.recommendedStudy.length > 0) {
        html += `<div class="recommended-study-section">
          <div class="section-title">Factors Suggested For Further Study</div>`;
        
        data.recommendedStudy.forEach(study => {
          // Capitalize first letter of definition and ensure it ends with a period
          let definition = study.definition;
          if (definition) {
            definition = definition.charAt(0).toUpperCase() + definition.slice(1);
            if (!definition.endsWith('.')) {
              definition += '.';
            }
          }
          
          html += `
            <div class="study-factor">
              <div class="study-factor-text">
                <span class="factor-name">${study.factor}</span> - ${definition}
              </div>
            </div>
          `;
        });
        
     html += `</div>`;
        
// Add reference code at the end
if (data.smi) {
  html += `
    <div style="text-align: right; font-size: 8pt; color: #999; margin-top: 30px; padding-top: 10px; border-top: 1px solid #eee;">
      Reference: 15439${data.smi.toFixed(2)}
    </div>
  `;
}
        
html += `
    </div>
  </div>
`;

return html;
 
	 
    }

    document.getElementById("artForm").addEventListener("submit", async function (event) {
      event.preventDefault();
      const artTitle = document.getElementById("artTitle").value.trim();
      const artistName = document.getElementById("artistName").value.trim();
      const subjectPhrase = document.getElementById("subjectPhrase").value.trim();
      const medium = document.getElementById("medium").value; 	  
      const imageInput = document.getElementById("image");
      const imageFile = imageInput.files[0];
      const responseContainer = document.getElementById("responseContainer");
      const downloadBtn = document.getElementById("downloadBtn");
      
      responseContainer.innerHTML = "";
      responseContainer.classList.remove('has-content');
      downloadBtn.style.display = "none";

      // Validate inputs
      if (!artTitle) {
        showCustomAlert("Please enter the artwork title.");
        return;
      }

      if (!artistName) {
        showCustomAlert("Please enter the artist name.");
        return;
      }

      if (!subjectPhrase) {
        showCustomAlert("Please describe the main subject and setting.");
        return;
      }


      if (!medium) {
        showCustomAlert("Please select a medium.");
        return;
      }



      if (!imageFile) {
        showCustomAlert("Please select an image file.");
        return;
      }
      
      // Enhanced image validation
      const validImageTypes = ['image/jpeg', 'image/jpg', 'image/png'];
      if (!validImageTypes.includes(imageFile.type.toLowerCase())) {
        showCustomAlert("Please upload a valid image file (JPEG or PNG only).");
        return;
      }
      
      const maxSizeMB = 50;
      if (imageFile.size > maxSizeMB * 1024 * 1024) {
        showCustomAlert(`File is too large. Maximum size is ${maxSizeMB}MB.`);
        return;
      }
      
      // Fetch prompt with timeout
      const controller1 = new AbortController();
      const timeoutId1 = setTimeout(() => controller1.abort(), 30000); // 30 second timeout
      
      let prompt;
      try {
        const promptResponse = await fetch("https://valora-analytics-api.onrender.com/prompts/ART_ANALYSIS", {
          signal: controller1.signal
        });
        clearTimeout(timeoutId1);
        
        if (!promptResponse.ok) {
          throw new Error("Failed to load prompt file");
        }
        prompt = await promptResponse.text();
      } catch (err) {
        clearTimeout(timeoutId1);
        
        if (err.name === 'AbortError') {
          showCustomAlert("Request timed out while loading prompt. Please try again.");
        } else {
          showCustomAlert("Could not load prompt file. Please try again.");
        }
        return;
      }
      
      responseContainer.innerHTML = `<p class="loading">Analyzing... please wait</p>`;
      responseContainer.classList.add('has-content');
      
      const convertToBase64 = (file) => {
        return new Promise((resolve, reject) => {
          const reader = new FileReader();
          reader.onload = () => resolve(reader.result.split(",")[1]);
          reader.onerror = (error) => {
            console.error("FileReader error:", error);
            reject(new Error("Failed to read image file. Please try a different image."));
          };
          reader.readAsDataURL(file);
        });
      };

      const convertImageToBase64 = (file) => {
        return new Promise((resolve, reject) => {
          const reader = new FileReader();
          reader.onload = () => resolve(reader.result); // Keep the full data URL with prefix
          reader.onerror = (error) => {
            console.error("FileReader error:", error);
            reject(new Error("Failed to process image for display. Please try a different image."));
          };
          reader.readAsDataURL(file);
        });
      };
      
      try {
        const base64Image = await convertToBase64(imageFile);
        const backendUrl = "https://valora-analytics-api.onrender.com/analyze-art";
        
        // Construct the prompt with artwork information and subject phrase
        const finalPrompt = `For context, artwork's subject and setting: ${subjectPhrase}

Title: "${artTitle}"
Artist: "${artistName}"

${prompt}`;

        const requestBody = { 
          prompt: finalPrompt, 
          image: base64Image,
          artTitle: artTitle,
          artistName: artistName,
          subjectPhrase: subjectPhrase,
		  medium: medium
        };
        
        // Analysis with timeout
        const controller2 = new AbortController();
        const timeoutId2 = setTimeout(() => controller2.abort(), 120000); // 120 second timeout
        
        const response = await fetch(backendUrl, {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify(requestBody),
          signal: controller2.signal
        });
        
        clearTimeout(timeoutId2);
        
        const result = await response.json();
        
        if (!response.ok) {
          throw new Error(`API Error: ${result.error?.message || "Unknown error"}`);
        }
        
        // Check if we got JSON structured data or legacy text format
        if (result.isLegacyFormat || result.analysis) {
          // Handle legacy format (fallback)
          showCustomAlert("Received legacy format. Please contact support to ensure backend is updated for JSON output.");
          responseContainer.innerHTML = "";
          responseContainer.classList.remove('has-content');
          return;
        }
        
        // Validate JSON structure
        if (!result.overview && !result.strengths && !result.opportunities) {
          throw new Error("Invalid response format from backend");
        }
        
        // Clean the result object - remove any SMI references
        const cleanResult = {
          title: result.title || "Analysis: 33 Essential Factors",
          artTitle: result.artTitle || artTitle,
          artistName: result.artistName || artistName,
          subjectPhrase: result.subjectPhrase || subjectPhrase,
          overview: result.overview,
          strengths: result.strengths || [],
          opportunities: result.opportunities || [],
          recommendedStudy: result.recommendedStudy || [],
		  smi: result.smi || null
        };
        
        // Create image URL for display
        const imageUrl = URL.createObjectURL(imageFile);
        
        // Generate timestamp
        const now = new Date();
        const dateOptions = { 
          year: 'numeric', 
          month: 'numeric', 
          day: 'numeric'
        };
        const timestamp = now.toLocaleString('en-US', dateOptions);
        
        // Build clean HTML from JSON data
        const analysisHTML = buildAnalysisHTML(cleanResult, imageUrl, timestamp);
        
        // Store data for PDF generation
        window.analysisData = {
          ...cleanResult,
          imageUrl: imageUrl,
          imageFile: imageFile,
          timestamp: timestamp
        };
        
        // Display the analysis
        responseContainer.innerHTML = analysisHTML;
        responseContainer.classList.add('has-content');
        
        // Clean up is not needed for base64 images
        
        downloadBtn.style.display = "block";
        
      } catch (error) {
        console.error("Error:", error);
        
        if (error.name === 'AbortError') {
          showCustomAlert("Analysis request timed out. Please try again.");
        } else {
          showCustomAlert(`Error: ${error.message}`);
        }
        
        responseContainer.innerHTML = "";
        responseContainer.classList.remove('has-content');
      }
    });

    function showCustomAlert(message) {
      const overlay = document.createElement('div');
      overlay.className = 'custom-alert-overlay';
      
      const alertBox = document.createElement('div');
      alertBox.className = 'custom-alert-box';
      
      const messagePara = document.createElement('p');
      messagePara.textContent = message;
      alertBox.appendChild(messagePara);
      
      const okButton = document.createElement('button');
      okButton.textContent = 'OK';
      okButton.onclick = () => {
        document.body.removeChild(overlay);
      };
      alertBox.appendChild(okButton);
      
      overlay.appendChild(alertBox);
      document.body.appendChild(overlay);
    }

    // PDF generation - copied from Calculate SMI method
    document.getElementById("downloadBtn").addEventListener("click", function() {
      const isMobilePhone = /iPhone|Android/i.test(navigator.userAgent) && !/iPad/i.test(navigator.userAgent);
      
      if (isMobilePhone) {
        showCustomAlert("Please use a desktop or tablet to download the PDF report.");
        return;
      }
      
      const element = document.getElementById("analysisText");
      if (!element) {
        showCustomAlert("No analysis content found. Please run analysis first.");
        return;
      }
      
      const originalMaxHeight = element.style.maxHeight;
      element.style.maxHeight = "none";

      // Temporarily change date color for PDF generation
      const dateElement = document.getElementById("resultDate");
      const originalDateColor = dateElement ? dateElement.style.color : '';
      if (dateElement) dateElement.style.color = "#000000";
      
      // Simplify image handling for PDF generation
      const images = element.querySelectorAll('img');
      Promise.all(Array.from(images).map(img => {
        if (img.complete) return Promise.resolve();
        return new Promise(resolve => {
          img.onload = resolve;
          img.onerror = resolve;
        });
      })).then(() => {
        // Use Calculate SMI PDF settings exactly
        const opt = {
          margin: 0.75,  // Smaller margins
          filename: `Analysis_${document.getElementById("artTitle").value.replace(/\s+/g, '_')}.pdf`,
          image: { type: 'jpeg', quality: 0.98 }, // Higher quality
          html2canvas: { 
            scale: 2, // Higher scale for better resolution
            useCORS: true, // Try to load cross-origin images
            logging: false,
            letterRendering: true,
            backgroundColor: "#ffffff"
          },
          jsPDF: { unit: 'in', format: 'letter', orientation: 'portrait' }
        };
        
        // Show a loading message
        const downloadBtn = document.getElementById("downloadBtn");
        const originalText = downloadBtn.textContent;
        downloadBtn.textContent = "Generating PDF...";
        downloadBtn.disabled = true;
        
        // Generate the PDF and save to downloads (Calculate SMI style)
        html2pdf().set(opt).from(element).save().then(() => {
          // Reset the button
          element.style.maxHeight = originalMaxHeight;
          if (dateElement) dateElement.style.color = originalDateColor;
          downloadBtn.textContent = originalText;
          downloadBtn.disabled = false;
          // Memory cleanup after PDF generation
          if (window.gc) window.gc();
        }).catch(err => {
          console.error("PDF generation error:", err);
          showCustomAlert("PDF generation failed. Please try again.");
          element.style.maxHeight = originalMaxHeight;
          if (dateElement) dateElement.style.color = originalDateColor;
          downloadBtn.textContent = originalText;
          downloadBtn.disabled = false;
        });
      });
    });

    // Clear form when title changes
    document.getElementById("artTitle").addEventListener("input", function () {
      document.getElementById("artistName").value = "";

      document.getElementById("image").value = "";
      
      const responseContainer = document.getElementById("responseContainer");
      responseContainer.innerHTML = "";
      responseContainer.classList.remove('has-content');
      
      document.getElementById("downloadBtn").style.display = "none";
    });
